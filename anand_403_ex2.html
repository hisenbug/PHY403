N = 10000
n = [1, 2, 5, 10, 50]
al = [0.5]*len(n)      # Can change alpha if needed
s = [0]*N
for i in n:
    for k in range(N):
        y = np.random.uniform(0, 1, size=i)
        s[k] = np.mean(y)
    plt.hist(s, bins=100, alpha=0.5, label="n = %d" %(i))
    # histtype : {'bar', 'barstacked', 'step', 'stepfilled'}

plt.title("Demonstrating the Central Limit Theorem")
plt.xlabel("Average Value")
plt.ylabel('#')
plt.gcf().set_size_inches(10, 6)
plt.legend(loc='best')
plt.show()

# 2.3: MC Integrator
# Note: I have not tuned my code for functions which can take -ve values.
def func(x):
    return (np.sin(x))**2 * np.exp(-x/3) * (np.cos(x))**8
    
x_min, x_max = np.pi/4, np.pi    # Limits
N = 100000                       # Sampling Points
f_vals = [0]*N
u = [0]*N
f_max= 0
ans = 0
count = 0

x = np.random.uniform(x_min, x_max, size=N)
f_vals = func(x)
f_max = np.max(f_vals)
u = np.random.uniform(0, f_max, size=N)
for m in range(N):
    if u[m] < f_vals[m]:
        count += 1
ans = count*(x_max - x_min)*f_max/N
print('2.3 The integral of given function from %.4g to %.4g is %.5g' %(x_min, x_max, ans))
