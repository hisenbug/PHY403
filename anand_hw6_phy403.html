def nlogP(par, xdata):
	"""Calculate the negative log probability """
	nu, k = par
	Di = nu * np.exp(-E0*xdata[:,0]/k)
	return -np.sum(xdata[:,1]*np.log(Di) - Di - gammaln(xdata[:,1]+1))

p0 = [1900., 1.20] # nu, k initial guess

## Problem 2
A, B = res.x[1]*0.75, res.x[1]*1.25
C, D = res.x[0]*0.75, res.x[0]*1.25

def lnprob(par, xdata):
	"""Calculate the negative log probability """
	nu, k = par
	lp = 0
	if nu < D and nu > C and k > A and k < B:
		lp = 0.0
	else:
		lp = -np.inf
	if not np.isfinite(lp):
		return -np.inf
	Di = nu * np.exp(-E0*xdata[:,0]/k)
	ans = -lp + np.sum(xdata[:,1]*np.log(Di) - Di - gammaln(xdata[:,1]+1))
	return ans

def lnlikej(par, xdata):
	"""Calculate the negative log probability """
	nu, k = par
	lp = 0
	if nu < D and nu > C and k > A and k < B:
		lp = np.log(1/(k*np.log(B/A)) )
	else:
		lp = -np.inf
	if not np.isfinite(lp):
		return -np.inf
	Di = nu * np.exp(-E0*xdata[:,0]/k)
	ans = lp + np.sum(xdata[:,1]*np.log(Di) - Di - gammaln(xdata[:,1]+1))
	return ans

def prior_transform(x):
	return np.array( [(D-C)*x[0]+C, A*((B/A)**x[1]) ] )
