import numpy as np
from scipy.stats import poisson
import matplotlib as mpl
from scipy import optimize
import matplotlib.pyplot as plt
from scipy.integrate import cumtrapz

def findConfidenceLevel(x, pdf, alpha):
	"""Given a PDF p(x), calculate the CDF and invert it to find quantiles.
	:x: array of independent values
	:pdf: PDF array p(x)
	:alpha: num or array with quantiles of the CDF [0..1]"""
	cdf = cumtrapz(pdf, x)
	return np.interp(alpha, cdf, x[:-1])

def my_function(xdata, para):
	if para[0] < 0 or para[1] < 0:
		return 0
	E_avg = 0.5*(xdata[:,1]+xdata[:,2])
	return para[0] + para[1]*np.exp(-0.5*(E_avg-E_0)**2/sigma2)

def NegLogLhood(para, xdata):
	lam = my_function(xdata, para)
	logl = np.sum(xdata[:,3]*np.log(lam) - lam)
	return -logl
